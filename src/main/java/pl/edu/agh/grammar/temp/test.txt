melody silnia(int n){
    int b = 0;

    if (n == 1){
        RETURN 1;
    }else{
        int a = 0;
        PLAY silnia(n-1) OUT a;
        RETURN n * a;
    }
}

melody gamma2(Note n){

melody gama(Note begin, Note finish, bool reverse){
    PLAY begin 1000;
    if(begin==finish){
        RETURN;
    }
    else{
        if(reverse==true){
            begin-=2;
        }
        else{
            Note e = C2;
            begin+=2;
            PLAY gama3();
            PLAY e 1000;
        }
        PLAY gama(begin,finish,reverse);
    }
}
