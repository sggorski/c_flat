melody gama(Note begin, Note finish, bool reverse){
    PLAY begin 1000;
    if(begin==finish){
        RETURN;
    }
    else{
        if(reverse==true){
            begin-=2;
        }
        else{
            begin+=2;
        }
        PLAY gama(begin,finish,reverse);
    }

}


melody main(){
    PLAY gama(C2,D3,false);
    PLAY gama(C3,C2,true);
}
