melody gama(Note begin, Note finish, bool reverse){
    PLAY begin 1000;
    if(begin==finish){
        RETURN;
    }
    else{
        if(reverse==true){
            begin-=2;
        }
        else{
            begin+=2;
        }
        PLAY gama(begin,finish,reverse);
    }
}

melody gama2(){
    PLAY C2 1000;
    PLAY gama(C1,2);
}
