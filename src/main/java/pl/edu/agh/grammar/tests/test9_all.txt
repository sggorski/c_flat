melody mel1(Note n1, int dist) {
	SET DISTORTION = dist;
	SET INTRUMENT = VIOLIN;

	PLAY n1 100;
	Note n2 = n1 + 4;
	PLAY n2 200;
	
	SET INSTRUMENT = PIANO;

	int i=1;
	while() {
		PLAY E1 100;
		PLAY F1 100;
		PLAY F#1 100;
		PLAY F1 100;
		i++;
		if(i==5) {
			break;
		}
	}
}

melody main() {
	SET INSTRUMENT = HARP;
	SET PACE = 20;

	PLAY H2 200;
	PAUSE 200;
	PLAY H2 100;
	Note n1 = H2;

	for(int i=0; i<10; i+=1) {
		PLAY n1 200;
		n1 += 2;
	}

	PAUSE 500;

	PLAY mel1(H1, 15);

	PAUSE 200;
	PLAY F#0 50;
	PLAY H1 200;

	Track t1;
	t1 ADD mel1(C1, 20);
	PLAY t1;
}
